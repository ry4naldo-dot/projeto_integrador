create table empresas(
  id int primary key auto_increment not null,
  nome_empresa varchar(255) not null,
  senha varchar(255) not null,
  tipo ENUM(
    'Eletronicos', 'Jogos', 'Seguranca', 
    'Diversao', 'Tecnologia', 'Mercado', 
    'Agronomia', 'Medicina', 'Ambiental', 
    'Hidroeletrica'
    ) NOT NULL,
  descricao text not null,
  numero varchar(50) not null,
  email varchar(255) unique not null,
  endereco varchar(255) not null
);

create table usuarios(
  id int primary key auto_increment not null,
  nome varchar(255) not null,
  senha varchar(255) not null,
  email varchar(255) unique not null
);

create table vagas(
  id int primary key auto_increment not null,
  id_empresas int not null,
  descricao text not null,
  tipo enum(
    'Eletronicos', 'Jogos', 'Seguranca', 
    'Diversao', 'Tecnologia', 'Mercado', 
    'Agronomia', 'Medicina', 'Ambiental', 
    'Hidroeletrica'
    ) not null,
  valor decimal(10, 2) not null,
  modelo ENUM('presencial', 'remoto', 'hibrido') NOT NULL,
  foreign key (id_empresas) references empresas(id)
);

create table curriculos(
  id int primary key auto_increment not null,
  id_usuarios int not null,
  id_vagas int not null,
  nome varchar(255) not null,
  descricao text not null,
  telefone varchar(50) not null,
  data_nascimento date not null,
  email varchar(255) not null,
  img text not null,
  tipo enum(
    'Eletronicos', 'Jogos', 'Seguranca', 
    'Diversao', 'Tecnologia', 'Mercado', 
    'Agronomia', 'Medicina', 'Ambiental', 
    'Hidroeletrica'
    ) not null,
  foreign key (id_usuarios) references usuarios(id),
  foreign key (id_vagas) references vagas(id)
);
