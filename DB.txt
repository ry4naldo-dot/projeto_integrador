-- Tabela: empresas

CREATE TABLE `empresas` (
    `id` INT UNSIGNED NOT NULL AUTO_INCREMENT,
    `nome_empresa` VARCHAR(255) NOT NULL,
    `senha` VARCHAR(255) NOT NULL,
    `tipo` ENUM(
    'Eletronicos', 'Jogos', 'Seguranca', 
    'Diversao', 'Tecnologia', 'Mercado', 
    'Agronomia', 'Medicina', 'Ambiental', 
    'Hidroeletrica'
    ) NOT NULL,
    `descricao` TEXT NOT NULL,
    `numero` VARCHAR(50) NOT NULL,
    `email` VARCHAR(255) NOT NULL,
    `endereco` VARCHAR(255) NOT NULL,
    PRIMARY KEY (`id`),
    UNIQUE KEY `idx_empresas_email` (`email`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Tabela: usuarios

CREATE TABLE `usuarios` (
    `id` INT UNSIGNED NOT NULL AUTO_INCREMENT,
    `nome` VARCHAR(255) NOT NULL,
    `senha` VARCHAR(255) NOT NULL,
    `endereco` VARCHAR(255) NOT NULL,
    `telefone` VARCHAR(50) NOT NULL,
    `email` VARCHAR(255) NOT NULL,
    `descricao` TEXT NOT NULL,
    `especialidades` TEXT NOT NULL,
    PRIMARY KEY (`id`),
    UNIQUE KEY `idx_usuarios_email` (`email`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Tabela: vagas

CREATE TABLE `vagas` (
    `id` INT UNSIGNED NOT NULL AUTO_INCREMENT,
    `id_usuarios` INT UNSIGNED NOT NULL,
    `id_empresas` INT UNSIGNED NOT NULL,
    `descricao` TEXT NOT NULL,
    `tipo` ENUM(
    'Eletronicos', 'Jogos', 'Seguranca', 
    'Diversao', 'Tecnologia', 'Mercado', 
    'Agronomia', 'Medicina', 'Ambiental', 
    'Hidroeletrica'
    ) NOT NULL,
    `valor` DECIMAL(10, 2) NOT NULL,
    `modelo` ENUM('presencial', 'remoto', 'hibrido') NOT NULL,
    PRIMARY KEY (`id`),
    INDEX `idx_vagas_id_usuarios` (`id_usuarios`),
    INDEX `idx_vagas_id_empresas` (`id_empresas`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Tabela: curriculos

    CREATE TABLE `curriculos` (
    `id` INT UNSIGNED NOT NULL AUTO_INCREMENT,
    `id_usuarios` INT UNSIGNED NOT NULL,
    `nome` VARCHAR(255) NOT NULL,
    `descricao` TEXT NOT NULL,
    `data_nascimento` DATE NOT NULL,
    `telefone` VARCHAR(50) NOT NULL,
    `img` TEXT NOT NULL,
    `email` VARCHAR(255) NOT NULL,
    `tipo` ENUM(
    'Eletronicos', 'Jogos', 'Seguranca', 
    'Diversao', 'Tecnologia', 'Mercado', 
    'Agronomia', 'Medicina', 'Ambiental', 
    'Hidroeletrica'
    ) NOT NULL,
    PRIMARY KEY (`id`),
    INDEX `idx_curriculos_id_usuarios` (`id_usuarios`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
